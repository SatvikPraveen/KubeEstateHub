# Location: `/helm-charts/kubeestatehub/templates/NOTES.txt`

1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.listingsApi.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "kubeestatehub.fullname" . }}-listings-api)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.listingsApi.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "kubeestatehub.fullname" . }}-listings-api'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "kubeestatehub.fullname" . }}-listings-api --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.listingsApi.service.port }}
{{- else if contains "ClusterIP" .Values.listingsApi.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "kubeestatehub.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component=listings-api" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Test the API endpoints:
{{- if .Values.ingress.enabled }}
  curl -X GET "{{ if .Values.ingress.tls }}https{{ else }}http{{ end }}://{{ (index .Values.ingress.hosts 1).host }}/api/v1/listings"
{{- else }}
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "kubeestatehub.fullname" . }}-listings-api 8080:80
  curl -X GET "http://localhost:8080/api/v1/listings"
{{- end }}

3. Access the Frontend Dashboard:
{{- if .Values.ingress.enabled }}
  Open your browser to: {{ if .Values.ingress.tls }}https{{ else }}http{{ end }}://{{ (index .Values.ingress.hosts 0).host }}
{{- else }}
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "kubeestatehub.fullname" . }}-frontend-dashboard 3000:80
  Open your browser to: http://localhost:3000
{{- end }}

{{- if .Values.monitoring.enabled }}

4. Access Monitoring:
{{- if .Values.monitoring.prometheus.enabled }}
  - Prometheus: kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "kubeestatehub.fullname" . }}-prometheus 9090:9090
    Then visit: http://localhost:9090
{{- end }}
{{- if .Values.monitoring.grafana.enabled }}
  - Grafana: kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "kubeestatehub.fullname" . }}-grafana 3000:3000
    Then visit: http://localhost:3000
    Default credentials: admin / {{ .Values.monitoring.grafana.adminPassword | default "admin" }}
{{- end }}
{{- end }}

{{- if .Values.imageStore.enabled }}

5. Access MinIO (Image Store):
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "kubeestatehub.fullname" . }}-image-store 9001:9001
  Then visit: http://localhost:9001
  Credentials: {{ .Values.imageStore.auth.rootUser }} / {{ .Values.imageStore.auth.rootPassword }}
{{- end }}

{{- if .Values.postgresql.enabled }}

6. Connect to PostgreSQL:
  export POSTGRES_PASSWORD=$(kubectl get secret --namespace {{ .Release.Namespace }} {{ include "kubeestatehub.fullname" . }}-db-secret -o jsonpath="{.data.password}" | base64 -d)
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "kubeestatehub.fullname" . }}-postgresql 5432:5432
  PGPASSWORD="$POSTGRES_PASSWORD" psql --host 127.0.0.1 --port 5432 --username {{ .Values.global.postgresql.auth.username }} --dbname {{ .Values.global.postgresql.auth.database }}
{{- end }}

7. Run tests:
  helm test {{ .Release.Name }} --namespace {{ .Release.Namespace }}

8. Check application status:
  kubectl --namespace {{ .Release.Namespace }} get all -l app.kubernetes.io/instance={{ .Release.Name }}

{{- if not .Values.ingress.enabled }}
⚠️  Warning: Ingress is disabled. The application will only be accessible via port-forwarding.
{{- end }}

{{- if not .Values.monitoring.enabled }}
⚠️  Warning: Monitoring is disabled. Consider enabling it for production deployments.
{{- end }}
